9. Program to print the largest element in an array
class main {
    public static void main(String[] args) {
        int[] array = {10, 20, 30, 40, 50, 60};

        int largest = array[0];
        
        for (int i = 1; i < array.length; i++) {
            if (array[i] > largest) {
                largest = array[i];
            }
        }

        System.out.println("Largest element in the array: " + largest);
    }
}

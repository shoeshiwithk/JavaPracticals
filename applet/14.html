14. Create an applet to display a digital clock.
import java.applet.Applet;
import java.awt.Color;
import java.awt.Font;
import java.awt.Graphics;
import java.util.Calendar;
import java.util.Timer;
import java.util.TimerTask;

class DigitalClockApplet extends Applet {
    private String time;

    public void init() {
        setBackground(Color.BLACK);
        setForeground(Color.WHITE);
        Timer timer = new Timer();
        timer.schedule(new TimerTask() {
            public void run() {
                repaint();
            }
        }, 0, 1000);
    }

    public void paint(Graphics g) {
        Calendar calendar = Calendar.getInstance();
        time = String.format("%02d:%02d:%02d", calendar.get(Calendar.
        HOUR_OF_DAY), calendar.get(Calendar.MINUTE), calendar
        .get(Calendar.SECOND));
        g.setFont(new Font("Arial", Font.BOLD, 48));
        g.drawString(time, 50, 100);
    }
}

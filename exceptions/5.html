5. Write a Java program that reads a file and throws an exception if the file is empty.
import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;

class main {
    public static void checkIfFileIsEmpty(String fileName) throws Exception {
        File file = new File(fileName);
        
        if (!file.exists() || file.length() == 0) {
            throw new Exception("The file is empty or does not exist.");
        }
        
        Scanner scanner = new Scanner(file);
        // Read the file to make sure it's not empty (we already checked file length)
        if (!scanner.hasNext()) {
            throw new Exception("The file is empty.");
        }

        while (scanner.hasNextLine()) {
            System.out.println(scanner.nextLine());
        }
        scanner.close();
    }

    public static void main(String[] args) {
        try {
            checkIfFileIsEmpty("numbers.txt");
        } catch (Exception e) {
            System.out.println("Caught an exception: " + e.getMessage());
        }
    }
}

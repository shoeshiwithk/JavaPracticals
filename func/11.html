10. Write a Java method that accepts three integers and checks whether they are consecutive or not. Returns true or false.
class main {
    public static void main(String[] args) {
        System.out.println(areConsecutive(3, 4, 5));  // true
        System.out.println(areConsecutive(10, 12, 11));  // true
        System.out.println(areConsecutive(1, 3, 5));  // false
    }

    public static boolean areConsecutive(int a, int b, int c) {
        int[] arr = {a, b, c};
        java.util.Arrays.sort(arr);
        return arr[2] - arr[1] == 1 && arr[1] - arr[0] == 1;
    }
}

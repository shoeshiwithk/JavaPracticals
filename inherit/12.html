12. Create a program, showing an example of super function
class Animal {
    public void sound() {
        System.out.println("Animal makes a sound");
    }
}

class Dog extends Animal {
    public void sound() {
        super.sound();
        System.out.println("Dog barks");
    }
}

class main {
    public static void main(String[] args) {
        Dog dog = new Dog();
        dog.sound();
    }
}

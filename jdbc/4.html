4. Update the row in the table in the database from Java API.
import java.sql.*;
class jdbc4{
public static void main(String s[]) throws java.lang.ClassNotFoundException,SQLException{
Class.forName("com.mysql.jdbc.Driver");
Connection c1=DriverManager.getConnection("jdbc:mysql://localhost:3306/madhu","root","root");
Statement st=c1.createStatement();
System.out.println(st.executeUpdate("Update student set name='Savita' where rollno=4"));
}}

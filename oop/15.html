15. Write a Java program to create a class called "MusicLibrary" with a collection of songs and methods to add and remove songs, and to play a random song.
import java.util.*;

class main {
    public static void main(String[] args) {
        MusicLibrary musicLibrary = new MusicLibrary();

        Song song1 = new Song("Shape of You", "Ed Sheeran");
        Song song2 = new Song("Blinding Lights", "The Weeknd");
        Song song3 = new Song("Levitating", "Dua Lipa");

        musicLibrary.addSong(song1);
        musicLibrary.addSong(song2);
        musicLibrary.addSong(song3);

        System.out.println("Music Library:");
        musicLibrary.displaySongs();

        musicLibrary.playRandomSong();

        musicLibrary.removeSong(song2);

        System.out.println("\nAfter removing 'Blinding Lights':");
        musicLibrary.displaySongs();

        musicLibrary.playRandomSong();
    }
}

class MusicLibrary {
    private List<Song> songs;

    public MusicLibrary() {
        songs = new ArrayList<>();
    }

    public void addSong(Song song) {
        songs.add(song);
    }

    public void removeSong(Song song) {
        songs.remove(song);
    }

    public void displaySongs() {
        if (songs.isEmpty()) {
            System.out.println("No songs in the library.");
        } else {
            for (Song song : songs) {
                System.out.println(song);
            }
        }
    }

    public void playRandomSong() {
        if (!songs.isEmpty()) {
            Random random = new Random();
            Song randomSong = songs.get(random.nextInt(songs.size()));
            System.out.println("\nPlaying: " + randomSong);
        } else {
            System.out.println("\nNo songs available to play.");
        }
    }
}

class Song {
    private String title;
    private String artist;

    public Song(String title, String artist) {
        this.title = title;
        this.artist = artist;
    }

    public String getTitle() {
        return title;
    }

    public String getArtist() {
        return artist;
    }

    @Override
    public String toString() {
        return title + " by " + artist;
    }
}

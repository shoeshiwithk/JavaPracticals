19. Write a Java program to create a class with methods to search for flights and hotels, and to book and cancel reservations.
import java.util.*;

class main {
    public static void main(String[] args) {
        TravelAgency travelAgency = new TravelAgency();

        travelAgency.searchFlights("New York", "Los Angeles");
        travelAgency.searchHotels("Los Angeles");

        travelAgency.bookFlight("Flight 101");
        travelAgency.bookHotel("Hotel California");

        travelAgency.cancelFlight("Flight 101");
        travelAgency.cancelHotel("Hotel California");
    }
}

class TravelAgency {
    private List<String> availableFlights;
    private List<String> availableHotels;
    private List<String> bookedFlights;
    private List<String> bookedHotels;

    public TravelAgency() {
        availableFlights = new ArrayList<>(Arrays.asList("Flight 101", "Flight 102", "Flight 103"));
        availableHotels = new ArrayList<>(Arrays.asList("Hotel California", "Hotel Sunset", "Hotel Grand"));
        bookedFlights = new ArrayList<>();
        bookedHotels = new ArrayList<>();
    }

    public void searchFlights(String from, String to) {
        System.out.println("Searching for flights from " + from + " to " + to + "...");
        for (String flight : availableFlights) {
            System.out.println("Available flight: " + flight);
        }
    }

    public void searchHotels(String location) {
        System.out.println("Searching for hotels in " + location + "...");
        for (String hotel : availableHotels) {
            System.out.println("Available hotel: " + hotel);
        }
    }

    public void bookFlight(String flight) {
        if (availableFlights.contains(flight)) {
            bookedFlights.add(flight);
            availableFlights.remove(flight);
            System.out.println("Flight " + flight + " booked successfully.");
        } else {
            System.out.println("Flight " + flight + " is not available.");
        }
    }

    public void cancelFlight(String flight) {
        if (bookedFlights.contains(flight)) {
            bookedFlights.remove(flight);
            availableFlights.add(flight);
            System.out.println("Flight " + flight + " cancelled successfully.");
        } else {
            System.out.println("No booking found for flight " + flight);
        }
    }

    public void bookHotel(String hotel) {
        if (availableHotels.contains(hotel)) {
            bookedHotels.add(hotel);
            availableHotels.remove(hotel);
            System.out.println("Hotel " + hotel + " booked successfully.");
        } else {
            System.out.println("Hotel " + hotel + " is not available.");
        }
    }

    public void cancelHotel(String hotel) {
        if (bookedHotels.contains(hotel)) {
            bookedHotels.remove(hotel);
            availableHotels.add(hotel);
            System.out.println("Hotel " + hotel + " cancelled successfully.");
        } else {
            System.out.println("No booking found for hotel " + hotel);
        }
    }
}

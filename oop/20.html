20. Create a class showing an example of default constructor.
class Person {
    String name;
    int age;

    public Person() {
        name = "John Doe";
        age = 30;
    }

    public void displayInfo() {
        System.out.println("Name: " + name);
        System.out.println("Age: " + age);
    }
}

class main {
    public static void main(String[] args) {
        Person person = new Person();
        person.displayInfo();
    }
}

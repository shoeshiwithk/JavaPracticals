6. Write a Java program to create a class called "Employee" with a name, job title, and salary attributes, and methods to calculate and update salary.
class main {
    public static void main(String[] args) {
        Employee employee1 = new Employee("Alice", "Software Engineer", 75000);
        Employee employee2 = new Employee("Bob", "Data Analyst", 65000);

        System.out.println("Before salary update:");
        System.out.println(employee1);
        System.out.println(employee2);

        employee1.updateSalary(10);
        employee2.updateSalary(5);

        System.out.println("\nAfter salary update:");
        System.out.println(employee1);
        System.out.println(employee2);
    }
}

class Employee {
    private String name;
    private String jobTitle;
    private double salary;

    public Employee(String name, String jobTitle, double salary) {
        this.name = name;
        this.jobTitle = jobTitle;
        this.salary = salary;
    }

    public String getName() {
        return name;
    }

    public String getJobTitle() {
        return jobTitle;
    }

    public double getSalary() {
        return salary;
    }

    public void updateSalary(double percentage) {
        double raise = salary * percentage / 100;
        salary += raise;
    }

    @Override
    public String toString() {
        return "Employee Name: " + name + ", Job Title: " + jobTitle + ", Salary: Rs" + salary;
    }
}

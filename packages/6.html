6. Create a Java program to create a package with 4 public classes, create the object in another file and call the function created in those classes.

Class A file
package mypackage;

public class ClassA {
    public void displayA() {
        System.out.println("This is Class A.");
    }
}


Class B file
package mypackage;

public class ClassB {
    public void displayB() {
        System.out.println("This is Class B.");
    }
}


Class C file
package mypackage;

public class ClassC {
    public void displayC() {
        System.out.println("This is Class C");
    }
}


Class D file
package mypackage;

public class ClassD {
    public void displayD() {
        System.out.println("This is Class D");
    }
}


Main file
import mypackage.ClassA;
import mypackage.ClassB;
import mypackage.ClassC;
import mypackage.ClassD;

class TestClasses {
    public static void main(String[] args) {
        ClassA objA = new ClassA();
        ClassB objB = new ClassB();
        ClassC objC = new ClassC();
        ClassD objD = new ClassD();
        
        objA.displayA();
        objB.displayB();
        objC.displayC();
        objD.displayD();
    }
}

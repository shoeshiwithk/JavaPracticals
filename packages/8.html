8. Create a Java program to create a package with public class and protected members to be accessed in another class.

Employee file
package company;

public class Employee {
    protected String name;
    protected int empId;
    protected double salary;
    public Employee(String name, int empId, double salary) {
        this.name = name;
        this.empId = empId;
        this.salary = salary;
    }
    protected void displayDetails() {
        System.out.println("Employee Name: " + name);
        System.out.println("Employee ID: " + empId);
        System.out.println("Salary: " + salary);
    }
}


Manager file
package company;

public class Manager extends Employee {
    public Manager(String name, int empId, double salary) {
        super(name, empId, salary);  // Call the parent constructor
    }
    public void showManagerDetails() {
        // Accessing protected method and members from Employee class
        displayDetails();  // Calling the inherited protected method
    }
}


Main file
package company;

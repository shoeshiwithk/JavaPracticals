9. Create a Java program to create a nested package i.e. c4.sample with public class and protected members to be accessed in another class.

Employee class
package c4.sample;

public class Employee {
    protected String name;
    protected int empId;
    protected double salary;
    public Employee(String name, int empId, double salary) {
        this.name = name;
        this.empId = empId;
        this.salary = salary;
    }
    protected void displayDetails() {
        System.out.println("Employee Name: " + name);
        System.out.println("Employee ID: " + empId);
        System.out.println("Salary: " + salary);
    }
}


Manager file
package c4.sample;

public class Manager extends Employee {
    public Manager(String name, int empId, double salary) {
        super(name, empId, salary); 
    }
    public void showManagerDetails() {
        displayDetails();  
    }
}


Main file
import c4.sample.Manager;

class main {
    public static void main(String[] args) {
        Manager manager = new Manager("Alice Smith", 102, 70000);
        manager.showManagerDetails();
    }
}

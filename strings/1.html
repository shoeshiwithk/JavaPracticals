1. Program to Check if a String contains only digits?
public class CheckStringDigits {
    public static void main(String[] args) {
        String str1 = "123456";
        String str2 = "1234a56";
        System.out.println(str1 + " contains only digits: " + containsOnlyDigits(str1));
        System.out.println(str2 + " contains only digits: " + containsOnlyDigits(str2));
    }

    public static boolean containsOnlyDigits(String str) {
        for (int i = 0; i < str.length(); i++) {
            if (!Character.isDigit(str.charAt(i))) {
                return false;
            }
        }
        return true;
    }
}

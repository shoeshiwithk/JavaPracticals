19. Program to divide a string in 'N' equal parts.
class main {
    public static void main(String[] args) {
        String str = "abcdefghij";
        int n = 5;
        
        if (str.length() % n == 0) {
            int partLength = str.length() / n;
            for (int i = 0; i < n; i++) {
                System.out.println(str.substring(i * partLength, (i + 1) * partLength));
            }
        } else {
            System.out.println("Cannot divide the string into " + n + " equal parts");
        }
    }
}

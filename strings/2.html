2. Program to perform Deep Copy for String?
class main {
    public static void main(String[] args) {
        String original = "Hello, World!";
        String deepCopy = new String(original);

        System.out.println("Original String: " + original);
        System.out.println("Deep Copy String: " + deepCopy);

        System.out.println("Are both strings different objects? " + (original != deepCopy));
    }
}

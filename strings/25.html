25. Program to find the duplicate words in a string
class main {
    public static void main(String[] args) {
        String str = "this is a test string and this is a test";
        String[] words = str.split("\\s+");
        java.util.Map<String, Integer> wordCount = new java.util.HashMap<>();
        
        for (String word : words) {
            wordCount.put(word, wordCount.getOrDefault(word, 0) + 1);
        }
        
        System.out.println("Duplicate words:");
        for (java.util.Map.Entry<String, Integer> entry : wordCount.entrySet()) {
            if (entry.getValue() > 1) {
                System.out.println(entry.getKey() + " appears " + entry.getValue() + " times");
            }
        }
    }
}
